program Simple
var
    a, count, tempVal : integer;
begin
    a := 10 + 5;          // a = 15
    count := a * 3;       // count = 45
    tempVal := (a + count) / 2; // tempVal = (15 + 45) / 2 = 30

    { 这是
      一个块注释 }
    
    // 这是单行注释
    a := tempVal - (count - a) * 2 + 1; // a = 30 - (45 - 15) * 2 + 1 = 30 - 30 * 2 + 1 = 30 - 60 + 1 = -29
end.
