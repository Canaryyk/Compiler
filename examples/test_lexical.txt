//测试用例

program TestProgram
    var i, sum, product : integer;
begin
    i := 1;
    sum := 0;
    product := 1;

    while i <= 10 do  //循环语句
    begin
        sum := sum + i;
        product := product * i;
        i := i + 1
    end;

    if sum > 50 then  //条件语句
        product := 0
    else
        sum := 0
end.
